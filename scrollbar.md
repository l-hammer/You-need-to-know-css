    justify-content: center;

  .scrollbar-main {

<template>
  <main>
    <div class="scrollbar-main">
      <span v-for="row in rows" class="row">{{ row }}</span>
    </div>
  </main>
</template>
<script>
  module.exports = {
    computed: {
      rows: function () {
        return Array.from({length: 56}).map((v, i) => Array(99).fill(i + 11).join(' '));
      }
    }
  }
</script>
</script>
